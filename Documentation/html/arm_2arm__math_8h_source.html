398  static __INLINE uint32_t arm_recip_q31(
399  q31_t in,
400  q31_t * dst,
401  q31_t * pRecipTable)
402  {
403 
404  uint32_t out, tempVal;
405  uint32_t index, i;
406  uint32_t signBits;
407 
408  if(in > 0)
409  {
410  signBits = __CLZ(in) - 1;
411  }
412  else
413  {
414  signBits = __CLZ(-in) - 1;
415  }
416 
417  /* Convert input sample to 1.31 format */
418  in = in << signBits;
419 
420  /* calculation of index for initial approximated Val */
421  index = (uint32_t) (in >> 24u);
422  index = (index & INDEX_MASK);
423 
424  /* 1.31 with exp 1 */
425  out = pRecipTable[index];
426 
427  /* calculation of reciprocal value */
428  /* running approximation for two iterations */
429  for (i = 0u; i < 2u; i++)
430  {
431  tempVal = (q31_t) (((q63_t) in * out) >> 31u);
432  tempVal = 0x7FFFFFFF - tempVal;
433  /* 1.31 with exp 1 */
434  out = (q31_t) (((q63_t) out * tempVal) >> 30u);
435  }
436 
437  /* write output */
438  *dst = out;
439 
440  /* return num of signbits of out = 1/in value */
441  return (signBits + 1u);
442 
443  }
