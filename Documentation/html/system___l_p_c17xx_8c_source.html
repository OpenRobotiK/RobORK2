424 void SystemCoreClockUpdate (void) /* Get Core Clock Frequency */
425 {
426  /* Determine clock frequency according to clock register values */
427  if (((LPC_SC->PLL0STAT >> 24) & 3) == 3) { /* If PLL0 enabled and connected */
428  switch (LPC_SC->CLKSRCSEL & 0x03) {
429  case 0: /* Int. RC oscillator => PLL0 */
430  case 3: /* Reserved, default to Int. RC */
431  SystemCoreClock = (IRC_OSC *
432  ((2ULL * ((LPC_SC->PLL0STAT & 0x7FFF) + 1))) /
433  (((LPC_SC->PLL0STAT >> 16) & 0xFF) + 1) /
434  ((LPC_SC->CCLKCFG & 0xFF)+ 1));
435  break;
436  case 1: /* Main oscillator => PLL0 */
437  SystemCoreClock = (OSC_CLK *
438  ((2ULL * ((LPC_SC->PLL0STAT & 0x7FFF) + 1))) /
439  (((LPC_SC->PLL0STAT >> 16) & 0xFF) + 1) /
440  ((LPC_SC->CCLKCFG & 0xFF)+ 1));
441  break;
442  case 2: /* RTC oscillator => PLL0 */
443  SystemCoreClock = (RTC_CLK *
444  ((2ULL * ((LPC_SC->PLL0STAT & 0x7FFF) + 1))) /
445  (((LPC_SC->PLL0STAT >> 16) & 0xFF) + 1) /
446  ((LPC_SC->CCLKCFG & 0xFF)+ 1));
447  break;
448  }
449  } else {
450  switch (LPC_SC->CLKSRCSEL & 0x03) {
451  case 0: /* Int. RC oscillator => PLL0 */
452  case 3: /* Reserved, default to Int. RC */
453  SystemCoreClock = IRC_OSC / ((LPC_SC->CCLKCFG & 0xFF)+ 1);
454  break;
455  case 1: /* Main oscillator => PLL0 */
456  SystemCoreClock = OSC_CLK / ((LPC_SC->CCLKCFG & 0xFF)+ 1);
457  break;
458  case 2: /* RTC oscillator => PLL0 */
459  SystemCoreClock = RTC_CLK / ((LPC_SC->CCLKCFG & 0xFF)+ 1);
460  break;
461  }
462  }
