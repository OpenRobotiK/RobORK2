76 uint32_t init_uart3(uint32_t baudrate )
77 {
78  uint32_t Fdiv;
79  uint32_t pclkdiv, pclk;
80 
81  LPC_PINCON->PINSEL0 &= ~0x0000000F;
82  LPC_PINCON->PINSEL0 |= 0x0000000A; /* RxD3 is P0.1 and TxD3 is P0.0 */
83  LPC_SC->PCONP |= 1<<4 | 1<<25; //Enable PCUART1
84  /* By default, the PCLKSELx value is zero, thus, the PCLK for
85  all the peripherals is 1/4 of the SystemFrequency. */
86  /* Bit 6~7 is for UART3 */
87  pclkdiv = (LPC_SC->PCLKSEL1 >> 18) & 0x03;
88  switch ( pclkdiv )
89  {
90  case 0x00:
91  default:
92  pclk = SystemCoreClock/4;
93  break;
94  case 0x01:
95  pclk = SystemCoreClock;
96  break;
97  case 0x02:
98  pclk = SystemCoreClock/2;
99  break;
100  case 0x03:
101  pclk = SystemCoreClock/8;
102  break;
103  }
104  LPC_UART3->LCR = 0x83; /* 8 bits, no Parity, 1 Stop bit */
105  Fdiv = ( pclk / 16 ) / baudrate ; /*baud rate */
106  LPC_UART3->DLM = Fdiv / 256;
107  LPC_UART3->DLL = Fdiv % 256;
108  LPC_UART3->LCR = 0x03; /* DLAB = 0 */
109  LPC_UART3->FCR = 0x07; /* Enable and reset TX and RX FIFO. */
110 
111  NVIC_EnableIRQ(UART3_IRQn);
112 
113  LPC_UART3->IER = IER_RBR | IER_THRE | IER_RLS; /* Enable UART3 interrupt */
114  return (TRUE);
115 }
